CREATE TABLE `log_documents` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `documentId` int(11) NOT NULL,
  `systemUserId` int(11) NOT NULL,
  `dateCreated` datetime NOT NULL,
  `actionPerformed` varchar(45) NOT NULL DEFAULT 'i',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


insert into log_documents 
(documentId, systemUserId, dateCreated)
select id, systemUserId, dateCreated from documents;
